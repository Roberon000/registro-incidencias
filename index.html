const nodemailer = require('nodemailer'); // Configurar transporte SMTP para Gmail const transporter = nodemailer.createTransport({ service: 'gmail', // Indica que usamos Gmail auth: { user: 'tu_correo@gmail.com', // Reemplaza con tu correo de Gmail pass: 'tu_contraseña_o_contraseña_de_aplicación' // Contraseña o contraseña de aplicación } }); // Configuración del correo const mailOptions = { from: 'tu_correo@gmail.com', // Remitente to: 'destinatario@correo.com', // Destinatario subject: 'Nueva Incidencia', // Asunto del correo text: 'Este es el contenido del correo.' // Contenido del correo }; // Enviar el correo transporter.sendMail(mailOptions, (error, info) => { if (error) { console.log('Error al enviar el correo:', error); } else { console.log('Correo enviado:', info.response); } });
